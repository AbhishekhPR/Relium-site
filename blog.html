-- before: this looks fine
SELECT c.id, COUNT(o.id) AS orders
FROM customers c
LEFT JOIN orders o ON c.id = o.customer_id
WHERE o.status = 'completed'

-- after: what you meant
LEFT JOIN orders o
ON c.id = o.customer_id
AND o.status = 'completed'
